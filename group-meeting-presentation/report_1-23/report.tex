    \begin{itemize}
        \item 对于一个k进制的n位加法需要的复合规则和子规则数
        \eee{
        \item 子规则数:\ff{4k^2}
        \item 进位规则数:\ff{2^{n+1}}
        \item 对齐规则数:\ff{2n^2}
        \item 运算规则数:\ff{2^n}
        } 
        合计大约：\ff{O(2^{n+1})}
        \item 模型:transformer
        \eee{
            \item 5层encode,decode
            \item 参数参数模型文件：300MB
        }
    \end{itemize}
\end{frame}
